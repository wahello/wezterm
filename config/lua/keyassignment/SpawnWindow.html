Wez's Terminal Emulator

SpawnWindow

Create a new window containing a tab from the default tab domain.

local wezterm = require 'wezterm';
return {
  keys = {
    {key="n", mods="SHIFT|CTRL", action="SpawnWindow"},
  }
}
